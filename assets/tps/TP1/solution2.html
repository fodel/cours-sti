Tâches à réaliser

Nous allons créer la base de données, définir les tables, leurs relations et insérer les données initiales.

1. Création de la base de données

Exécutez cette commande SQL dans votre SGBDR (phpMyAdmin, etc.) pour créer la base de données.

SQL
CREATE DATABASE `DCP2_votre_nom_prenom`;
2. Création des tables et ajout des clés

Voici le code SQL complet. La table reservations a une clé primaire composée de plusieurs champs, et les clés étrangères sont définies.

SQL
USE `DCP2_votre_nom_prenom`;

-- Table Citoyens
CREATE TABLE `Citoyens` (
  `cin` CHAR(8) NOT NULL,
  `nom` VARCHAR(50) NOT NULL,
  `prenom` VARCHAR(50) NOT NULL,
  `email` VARCHAR(100) NOT NULL,
  `telephone` VARCHAR(20) NOT NULL,
  `dateNaissance` DATE NOT NULL,
  `pin` CHAR(6) NOT NULL,
  PRIMARY KEY (`cin`)
);

-- Table centres
CREATE TABLE `centres` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `nom` VARCHAR(100) NOT NULL,
  `adresse` VARCHAR(255) NOT NULL,
  `capaciteMax` INT NOT NULL,
  PRIMARY KEY (`id`)
);

-- Table vaccins
CREATE TABLE `vaccins` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `nom` VARCHAR(50) NOT NULL,
  `fabricant` VARCHAR(50) NOT NULL,
  `stock` INT NOT NULL,
  PRIMARY KEY (`id`)
);

-- Table reservations avec une clé primaire composée et les clés étrangères
CREATE TABLE `reservations` (
  `cin` CHAR(8) NOT NULL,
  `iDcentre` INT NOT NULL,
  `iDvaccin` INT NOT NULL,
  `dateReservation` DATE NOT NULL,
  `statut` VARCHAR(20) NOT NULL DEFAULT 'en attente',
  PRIMARY KEY (`cin`, `iDcentre`, `iDvaccin`, `dateReservation`),
  FOREIGN KEY (`cin`) REFERENCES `Citoyens`(`cin`),
  FOREIGN KEY (`iDcentre`) REFERENCES `centres`(`id`),
  FOREIGN KEY (`iDvaccin`) REFERENCES `vaccins`(`id`)
);

mesControles.js

// Vérifie si une chaîne ne contient que des chiffres
function verifNum(chaine) {
    if (chaine.length === 0) return false;
    for (let i = 0; i < chaine.length; i++) {
        if (chaine.charAt(i) < '0' || chaine.charAt(i) > '9') {
            return false;
        }
    }
    return true;
}

// Vérifie si une chaîne ne contient que des lettres (non accentuées)
function verifAlpha(chaine) {
    if (chaine.length === 0) return false;
    for (let i = 0; i < chaine.length; i++) {
        let char = chaine.charAt(i).toUpperCase();
        if (char < 'A' || char > 'Z') {
            return false;
        }
    }
    return true;
}


// --- Fonctions principales de la page ---

// Génère un PIN à 6 chiffres et l'affiche dans le champ correspondant

function generatePin() {
    var pin = Math.trunc(100000 + Math.random() * 900000);
    document.getElementById('pin').value = pin;
}

// Vérifie la validité des champs du formulaire de réservation
function verif1() {
    var cin = document.getElementById('cin').value;
    var nom = document.getElementById('nom').value;
    var prenom = document.getElementById('prenom').value;
    var telephone = document.getElementById('telephone').value;
    var dateNaissance = document.getElementById('dateNaissance').value;
    var centre = document.getElementById('centre').value;
    var vaccin = document.getElementById('vaccin').value;
    var dateVaccination = document.getElementById('dateVaccination').value;

    // Contrôle CIN : 8 chiffres
    if (cin.length != 8 || !verifNum(cin)) {
        alert("Le CIN doit être une chaîne de 8 chiffres.");
        return false;
    }

    // Contrôle Nom : non vide, max 30, commence par une lettre
    if (nom.length === 0 || nom.length > 30 || !verifAlpha(nom.charAt(0))) {
        alert("Le nom est invalide.");
        return false;
    }

    // Contrôle Prénom : non vide, max 30, commence par une lettre
    if (prenom.length === 0 || prenom.length > 30 || !verifAlpha(prenom.charAt(0))) {
        alert("Le prénom est invalide.");
        return false;
    }

    // Contrôle Téléphone : 8 chiffres, premier > 1
    if (telephone.length != 8 || !verifNum(telephone) || telephone.charAt(0) <= '1') {
        alert("Le téléphone doit être une chaîne de 8 chiffres commençant par un chiffre > 1.");
        return false;
    }

    // Contrôle Date de naissance : obligatoire
    if (dateNaissance === "") {
        alert("La date de naissance est obligatoire.");
        return false;
    }

    // Contrôle Centre : obligatoire
    if (centre === "") {
        alert("Veuillez choisir un centre de vaccination.");
        return false;
    }

    // Contrôle Vaccin : obligatoire
    if (vaccin === "") {
        alert("Veuillez choisir un type de vaccin.");
        return false;
    }

    // Contrôle Date de vaccination : > date système
    var today = new Date();
    var vaccDate = new Date(dateVaccination);
    if (vaccDate <= today) {
        alert("La date de vaccination doit être supérieure à la date du jour.");
        return false;
    }

    return true; // Tout est bon, le formulaire peut être soumis
}

if ($stock <= 0) {
    die("Dose de vaccin épuisée !");

// 4. Si tout est bon, insérer les données

3. Fichier suivie.php

Ce script affiche les détails de la réservation si CIN et PIN sont valides.

suivie.php
<?php
// Connexion à la base de données
 $connexion = mysqli_connect("localhost", "root", "", "DCP2_votre_nom_prenom");

// Récupération des données
 $cin = $_POST['cin'];
 $pin = $_POST['pin'];

// 1. Vérifier si le CIN existe
 $checkCin = "SELECT * FROM Citoyens WHERE cin = '$cin'";
 $resultat = mysqli_query($connexion, $checkCin);
if (mysqli_num_rows($resultat) == 0) {
    die("CIN inexistant !");
}

// 2. Vérifier si le PIN est valide

 $citoyen = mysqli_fetch_array($resultat);
if ($citoyen['pin'] != $pin) {
    die("PIN invalide !");
}

// 3. Si tout est bon, afficher la fiche de vaccination
// Requête pour récupérer les informations de la réservation la plus proche
 $ficheQuery = "SELECT 
                c.nom, c.prenom, 
                ce.nom AS nom_centre, 
                v.nom AS nom_vaccin, 
                r.dateReservation, 
                r.statut
              FROM reservations r, Citoyens c, centres ce, vaccins v
              WHERE r.cin = c.cin
              AND r.iDcentre = ce.id
              AND r.iDvaccin = v.id
              AND r.cin = '$cin'
              AND r.dateReservation = (SELECT MAX(dateReservation) FROM reservations WHERE cin = '$cin')";

 $resFiche = mysqli_query($connexion, $ficheQuery);

 $fiche = mysqli_fetch_array($resFiche);

// Affichage des résultats
echo "Bienvenue " . $fiche['nom'] . " " . $fiche['prenom'] . "<br>";
echo "Votre vaccin le plus proche sera :<br>";
echo "<ul>";
echo "<li>Centre : " . $fiche['nom_centre'] . "</li>";
echo "<li>Vaccin : " . $fiche['nom_vaccin'] . "</li>";
echo "<li>Date : " . $fiche['dateReservation'] . "</li>";
echo "<li>Statut : " . $fiche['statut'] . "</li>";
echo "</ul>";

mysqli_close($connexion);
?>
